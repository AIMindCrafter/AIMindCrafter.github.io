from langchain.agents import initialize_agent, AgentType
from langchain.llms import OpenAI

def build_ai_agent():
    # Connect robust AI models with a single interface.
    llm = OpenAI(temperature=0.7)
    
    agent = initialize_agent(
        tools=[], 
        llm=llm, 
        agent=AgentType.ZERO_SHOT_REACT_DESCRIPTION, 
        verbose=True
    )
    
    return agent

agent_instance = build_ai_agent()
agent_instance.run("Analyze and summarize context via RAG pipeline.")
